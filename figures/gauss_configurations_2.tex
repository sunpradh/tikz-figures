\newcommand{\DrawConfiguration}[6]{
    \begin{scope}[xshift=#1, yshift=#2]
        \draw[step=2,black,thin] (-1,-1) grid (1,1);
        \node[#3] at ( 0.5,  0.0) {};
        \node[#4] at ( 0.0,  0.5) {};
        \node[#5] at (-0.5,  0.0) {};
        \node[#6] at ( 0.0, -0.5) {};
    \end{scope}
}

\end{tikzpicture}
