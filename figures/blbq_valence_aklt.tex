\begin{tikzpicture}[
        scale=0.75,
        spin one/.style={fill=Blue20, fill opacity=1, draw=black, thin, inner sep=0cm, minimum size=0.5cm, thick},
        spin half/.style={circle, fill=Orange, draw=black, inner sep=0cm, minimum size=0.35cm},
        entangled/.style={line width=2pt},
        nonentangled/.style={dotted, thick},
        font=\small
    ]
    % spin 1 sites

    }
